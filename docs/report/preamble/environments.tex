\definecolor{definition_background}{HTML}{fcf8f7}
\definecolor{definition_accent}{HTML}{730a05}
\definecolor{definition_title_mid_background}{HTML}{730a05}
\definecolor{definition_title_left_background}{HTML}{b52922}
\definecolor{definition_title_right_background}{HTML}{870c05}
\definecolor{definition_title_text}{HTML}{faf5f5}

\newtcolorbox{definition}[2][]
{enhanced,
	before skip=2mm,
    after skip=2mm,
    colback=definition_background,
    colframe=definition_accent,boxrule=0.5mm,
	attach boxed title to top left={xshift=1cm,yshift*=1mm-\tcboxedtitleheight},
    varwidth boxed title*=-3cm,
	boxed title style={
        frame code={
        \path[fill=definition_title_left_background]
            ([yshift=-1mm,xshift=-1mm]frame.north west)
            arc[start angle=0,end angle=180,radius=1mm]
            ([yshift=-1mm,xshift=1mm]frame.north east)
            arc[start angle=180,end angle=0,radius=1mm];
        \path[
            left color=definition_title_left_background!80!definition_accent,  
            right color=definition_title_right_background!60!definition_accent, 
            middle color=definition_title_mid_background!60!definition_accent
        ]
            ([xshift=-2mm]frame.north west) -- ([xshift=2mm]frame.north east)
            [rounded corners=1mm] -- ([xshift=1mm,yshift=-1mm]frame.north east)
            -- (frame.south east) -- (frame.south west)
            -- ([xshift=-1mm,yshift=-1mm]frame.north west)
            [sharp corners]-- cycle;
				},
        interior engine=empty,
	},
	fonttitle=\bfseries,
	coltitle=definition_title_text,
	title={#2},#1}

            [rounded corners=2mm] |-
            (frame.north) -| cycle;
        },
    #1
}

%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
% Note Environment
%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%

\usetikzlibrary{arrows,calc,shadows.blur}
\tcbuselibrary{skins}
\newtcolorbox{note}[2][]{%
    enhanced jigsaw,
    colback=gray!20!white,%
    colframe=gray!80!black,
    size=small,
    boxrule=1pt,
    title=\textbf{#2},
    halign title=flush center,
    coltitle=black,
    breakable,
    drop shadow=black!50!white,
    attach boxed title to top left={xshift=1cm,yshift=-\tcboxedtitleheight/2,yshifttext=-\tcboxedtitleheight/2},
    boxed title style={%
            colback=white,
            size=fbox,
            boxrule=1pt,
            boxsep=2pt,
            underlay={%
                    \coordinate (dotA) at ($(interior.west) + (-0.5pt,0)$);
                    \coordinate (dotB) at ($(interior.east) + (0.5pt,0)$);
                    \begin{scope}
                        \clip (interior.north west) rectangle ([xshift=3ex]interior.east);
                        \filldraw [
                            white,
                            blur shadow={shadow opacity=60, shadow yshift=-.75ex}, 
                            rounded corners=2pt
                        ] (interior.north west) rectangle (interior.south east);
                    \end{scope}
                    \begin{scope}[gray!80!black]
                        \fill (dotA) circle (0.5pt);
                        \fill (dotB) circle (0.5pt);
                    \end{scope}
                },
        },
    #1,
}
